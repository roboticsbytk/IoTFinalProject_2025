WiFiClientSecure net;
PubSubClient client(net);

void setup_wifi() {
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWiFi connected");
}

void setup() {
   Serial.begin(115200);
  setup_wifi();
  // while (WiFi.status() != WL_CONNECTED) delay(500);

  net.setCACert(ca_cert);  // trust EMQX broker certificate
  client.setServer("n12a20fa.ala.asia-southeast1.emqxsl.com", 8883);

  if (client.connect("ESP32ClientTLS", "test", "1234")) {
    client.publish("tahniat/esp32/out", "Hello secure EMQX world!");
  }
}

void loop() {
  if (!client.connected()) {
    client.connect("ESP32ClientTLS");
  }
  client.loop();

  static unsigned long lastMsg = 0;
  if (millis() - lastMsg > 5000) {
    lastMsg = millis();
    String msg = "Hello from ESP32 at " + String(millis()/1000) + "s";
    client.publish("tahniat/esp32/out", msg.c_str());
    Serial.println("Published: " + msg);
  }
}
